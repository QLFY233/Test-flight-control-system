# 试飞控制系统 — 全局实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 从零搭建完整的试飞控制系统——打通"大模型理解试飞意图 → 翻译为无人机目标点 → 求解器生成安全轨迹 → 模拟无人机执行"的完整链路。

**Architecture:** 四大子系统分层协同：(1) 前端纯静态 SPA(HTML/JS + Three.js 3D + ECharts 2D)，经 SSE/WS/REST 连接后端 A；(2) 后端 A(FastAPI + Pydantic AI, Python 3.10+)为 Agent 中枢，运行 α(翻译器)/β(中枢调度)双 Agent；(3) 后端 B(Python 3.8 + ROS Noetic)为实时飞控桥，含求解器与监控程序，经 Unix socket 接收 A 下发；(4) sim-drone 为阶段 1 假无人机 ROS 节点。α 后期可替换为小模型(独立训练管线)。

**Tech Stack:** FastAPI 0.136.3, Pydantic AI 2.0.0, SQLAlchemy 2.0.51, msgpack, ROS Noetic, rospy, Three.js, ECharts, DeepSeek-V4-Flash, 讯飞 STT/TTS

## Global Constraints

- **双虚拟环境**: venv-A(Python 3.10+) 与 venv-B(Python 3.8 + `--system-site-packages`)，不共享 site-packages，仅经 Unix socket(msgpack)通信
- **WSL2 Ubuntu 20.04 + ROS Noetic**: 已装 `/opt/ros/noetic`，mirrored 网络模式
- **安全边界**: 大模型绝不直接碰飞控；α 产出经求解器物理校验才下发；异常只告警不自动执行
- **API key 一律环境变量**: DeepSeek/讯飞密钥不入库不入 config
- **msgpack use_bin_type=True**: A/B 两侧必须一致，是 S0 联调第 1 验证项
- **先导阶段**: 单机、室内、单无人机、纯 ROS 假无人机(阶段 2 再切 PX4 SITL)
- **小模型 α 不阻塞先导**: 先导全用 LLM α，小模型训练管线在阶段二末启动
- **video_frame 先导期不实现**: 阶段 1 无摄像头，VideoPanel 占位"无信号"

```python
"""配置加载与 Pydantic 校验."""
from dataclasses import dataclass, field
from pathlib import Path
import yaml
from pydantic import BaseModel, ValidationError


class FieldBoundary(BaseModel):
    x: list[float]  # [min, max]
    y: list[float]
    z: list[float]


class HomeConfig(BaseModel):
    position: list[float]  # [x, y, z]
    yaw: float


class ObstacleConfig(BaseModel):
    id: int
    type: str
    center: list[float]
    # per-type params
    size: list[float] | None = None       # box
    radius: float | None = None           # cylinder/sphere/capsule
    height: float | None = None           # cylinder
    point: list[float] | None = None      # plane
    normal: list[float] | None = None     # plane
    p1: list[float] | None = None         # capsule
    p2: list[float] | None = None         # capsule
    label: str = ""


class FieldConfig(BaseModel):
    boundary: FieldBoundary
    home: HomeConfig
    obstacles: list[ObstacleConfig] = []


class GlobalConstraints(BaseModel):
    speed_max: float = 1.5
    accel_max: float = 2.0
    angular_velocity_max: float = 0.5
    keep_clear_distance: float = 0.5
    ceiling: float = 2.5
    floor: float = 0.3


class PresetConstraints(BaseModel):
    speed_max: float
    accel_max: float
    keep_clear_distance: float


class DefaultConstraints(BaseModel):
    global_: GlobalConstraints = field(default_factory=GlobalConstraints)
    presets: dict[str, PresetConstraints] = field(default_factory=dict)


@dataclass
class Config:
    field: FieldConfig
    constraints: DefaultConstraints
    alpha_loop_period: float = 2.0
    alpha_history_rounds: int = 10


def load_config(
    field_path: str = "config/field.yaml",
    constraints_path: str = "config/default_constraints.yaml",
) -> Config:
    try:
        with open(field_path) as f:
            field_raw = yaml.safe_load(f)
        field = FieldConfig(**field_raw)
    except (FileNotFoundError, ValidationError) as e:
        raise SystemExit(f"Failed to load field config: {e}")

    try:
        with open(constraints_path) as f:
            constraints_raw = yaml.safe_load(f)
        global_ = GlobalConstraints(**constraints_raw.get("global", {}))
        presets = {
            k: PresetConstraints(**v)
            for k, v in constraints_raw.get("presets", {}).items()
        }
        constraints = DefaultConstraints(global_=global_, presets=presets)
    except (FileNotFoundError, ValidationError) as e:
        raise SystemExit(f"Failed to load constraints config: {e}")

    import os
    return Config(
        field=field,
        constraints=constraints,
        alpha_loop_period=float(os.environ.get("ALPHA_LOOP_PERIOD", "2.0")),
        alpha_history_rounds=int(os.environ.get("ALPHA_HISTORY_ROUNDS", "10")),
    )
```

"""B 侧共享状态."""
from dataclasses import dataclass, field
from threading import Lock


@dataclass
class PoseData:
    x: float = 0.0
    y: float = 0.0
    z: float = 0.0
    qw: float = 1.0
    qx: float = 0.0
    qy: float = 0.0
    qz: float = 0.0


@dataclass
class BState:
    pose_lock: Lock = field(default_factory=Lock)
    _current_pose: PoseData | None = None
    ipc_connected: bool = False
    solver_status: str = "idle"  # idle/executing/hovering/aborted
    current_trajectory: object | None = None
    current_segment_index: int = 0
    last_data_ts: float = 0.0
    field: object | None = None
    default_constraints: dict = field(default_factory=dict)

    @property
    def current_pose(self) -> PoseData | None:
        with self.pose_lock:
            return self._current_pose

    @current_pose.setter
    def current_pose(self, val: PoseData | None):
        with self.pose_lock:
            self._current_pose = val
```

"""长度前缀帧编解码 (msgpack)."""
import struct
import msgpack
from backend_B.bus.protocol import MSGPACK_USE_BIN_TYPE

MAX_FRAME_SIZE = 16 * 1024 * 1024  # 16 MiB


async def send_frame(writer, msg: dict) -> None:
    payload = msgpack.packb(msg, use_bin_type=MSGPACK_USE_BIN_TYPE)
    writer.write(struct.pack(">I", len(payload)) + payload)
    await writer.drain()


async def recv_frame(reader) -> dict:
    header = await reader.readexactly(4)
    length = struct.unpack(">I", header)[0]
    if length > MAX_FRAME_SIZE:
        raise ValueError(f"Frame too large: {length}")
    payload = await reader.readexactly(length)
    return msgpack.unpackb(payload, raw=False, use_bin_type=MSGPACK_USE_BIN_TYPE)
```

- [ ] **Step 2: 编写同步版 (B 侧用 threading，不用 asyncio)**

# backend-B/ipc/frames.py (追加)
def send_frame_sync(sock: socket.socket, msg: dict) -> None:
    payload = msgpack.packb(msg, use_bin_type=MSGPACK_USE_BIN_TYPE)
    sock.sendall(struct.pack(">I", len(payload)) + payload)


def recv_frame_sync(sock: socket.socket) -> dict:
    header = _recv_exact(sock, 4)
    length = struct.unpack(">I", header)[0]
    if length > MAX_FRAME_SIZE:
        raise ValueError(f"Frame too large: {length}")
    payload = _recv_exact(sock, length)
    return msgpack.unpackb(payload, raw=False, use_bin_type=MSGPACK_USE_BIN_TYPE)


def _recv_exact(sock: socket.socket, n: int) -> bytes:
    buf = b""
    while len(buf) < n:
        chunk = sock.recv(n - len(buf))
        if not chunk:
            raise ConnectionError("Socket closed")
        buf += chunk
    return buf
```

#!/usr/bin/env python3
"""阶段1 假无人机: 运动学积分, 订阅 setpoint, 发布位姿/IMU."""
import rospy
from geometry_msgs.msg import PoseStamped, TwistStamped, Twist, Vector3
from sensor_msgs.msg import Imu
from collections import deque
import math
import time


class FakeDrone:
    def __init__(self):
        self.pos = [0.0, 0.0, 0.5]
        self.vel = [0.0, 0.0, 0.0]
        self.yaw = 0.0
        self.target = None
        self.last_setpoint_ts = time.time()
        self.max_speed = rospy.get_param("~max_speed", 2.0)
        self.timeout = rospy.get_param("~setpoint_timeout", 0.5)
        self.boundary = rospy.get_param("~boundary", {"x": [0,5], "y": [0,4], "z": [0,3]})

        self.pose_pub = rospy.Publisher("/drone/local_position/pose", PoseStamped, queue_size=10)
        self.vel_pub = rospy.Publisher("/drone/local_position/velocity", TwistStamped, queue_size=10)
        self.imu_pub = rospy.Publisher("/drone/imu/data", Imu, queue_size=10)
        rospy.Subscriber("/drone/setpoint_position/local", PoseStamped, self._setpoint_cb)

    def _setpoint_cb(self, msg: PoseStamped):
        self.target = [msg.pose.position.x, msg.pose.position.y, msg.pose.position.z]
        self.last_setpoint_ts = time.time()

    def _clamp_to_boundary(self, pos):
        b = self.boundary
        return [
            max(b["x"][0], min(b["x"][1], pos[0])),
            max(b["y"][0], min(b["y"][1], pos[1])),
            max(b["z"][0], min(b["z"][1], pos[2])),
        ]

    def update(self, dt: float):
        if self.target is None or (time.time() - self.last_setpoint_ts) > self.timeout:
            return  # 超时悬停: 维持当前位姿

        dx = self.target[0] - self.pos[0]
        dy = self.target[1] - self.pos[1]
        dz = self.target[2] - self.pos[2]
        dist = math.sqrt(dx*dx + dy*dy + dz*dz)

        if dist < 0.05:
            self.vel = [0.0, 0.0, 0.0]
            return

        speed = min(self.max_speed, dist / dt)
        self.vel = [speed * dx / dist, speed * dy / dist, speed * dz / dist]
        self.pos = [self.pos[i] + self.vel[i] * dt for i in range(3)]
        self.pos = self._clamp_to_boundary(self.pos)
        self.yaw = math.atan2(self.vel[1], self.vel[0]) if abs(self.vel[0]) > 1e-6 or abs(self.vel[1]) > 1e-6 else self.yaw

    def publish(self):
        t = rospy.Time.now()

        pose = PoseStamped()
        pose.header.stamp = t
        pose.header.frame_id = "map"
        pose.pose.position.x, pose.pose.position.y, pose.pose.position.z = self.pos
        cy, sy = math.cos(self.yaw * 0.5), math.sin(self.yaw * 0.5)
        pose.pose.orientation.w, pose.pose.orientation.z = cy, sy
        self.pose_pub.publish(pose)

        vel = TwistStamped()
        vel.header.stamp = t
        vel.twist.linear.x, vel.twist.linear.y, vel.twist.linear.z = self.vel
        self.vel_pub.publish(vel)

        imu = Imu()
        imu.header.stamp = t
        imu.linear_acceleration.x, imu.linear_acceleration.y, imu.linear_acceleration.z = 0.0, 0.0, 9.81
        imu.orientation = pose.pose.orientation
        self.imu_pub.publish(imu)


def main():
    rospy.init_node("fake_drone")
    drone = FakeDrone()
    rate = rospy.Rate(50)
    while not rospy.is_shutdown():
        drone.update(0.02)
        drone.publish()
        rate.sleep()


if __name__ == "__main__":
    main()
```

"""约束层级解析: 点 > 段 > 全局 > 默认."""
def resolve_constraints(
    point_c: dict | None = None,
    segment_c: dict | None = None,
    global_c: dict | None = None,
    default_c: dict | None = None,
) -> dict:
    merged = dict(default_c or {})
    if global_c:
        merged.update(global_c)
    if segment_c:
        merged.update(segment_c)
    if point_c:
        merged.update(point_c)
    return merged


KNOWN_KEYS = {
    "speed_max", "accel_max", "angular_velocity_max",
    "keep_clear_distance", "ceiling", "floor",
}


def check_unknown_keys(constraints: dict) -> list[str]:
    return [k for k in constraints if k not in KNOWN_KEYS]

import asyncio
from .registry import get as get_component
from .protocol import MSG_CALL, MSG_RESULT, MSG_ERROR

async def call(to: str, tool: str, args: dict, call_id: str = None) -> dict:
    comp = get_component(to)
    if comp is None:
        return {"msg_type": MSG_ERROR, "error": f"component '{to}' not found"}
    try:
        result = await comp.handle(tool, args)
        return {"msg_type": MSG_RESULT, "result": result}
    except Exception as e:
        return {"msg_type": MSG_ERROR, "error": str(e)}

async def alpha_loop(app_state, bus):
    while True:
        await asyncio.sleep(app_state.config.alpha_loop_period)
        pose = app_state.current_pose
        alpha_inputs = app_state.drain_alpha_input_queue()
        if alpha_inputs:
            spec = await translator.translate(alpha_inputs, pose)
            app_state.current_trajectory_plan = spec
        elif app_state.current_trajectory_plan and app_state.current_trajectory_plan.has_remaining():
            spec = app_state.current_trajectory_plan.next_segment()
        else:
            spec = TrajectorySpec.hover(pose)
        if spec:
            await bus.call(to="solver", tool="trajectory", args=spec.dict())
            await ws_push_alpha_output({"trajectory": spec})

# 依赖关系与建议实施顺序

```
Phase 0 (环境) ──> Phase 1 (配置/协议)
                       │
          ┌────────────┤
          ▼            ▼
    Phase 2 (B+sim)  Phase 3 (A core)   ← 可并行开发，按 spec 接口对齐
          │            │
          └─────┬──────┘
                ▼
         Phase 4 (A↔B 联调)
                │
                ▼
         Phase 5 (前端)   ← 可在 Phase 3 完成后启动，部分阶段可与 Phase 4 并行
                │
                ▼
         Phase 6 (小模型 α) ← 可在 Phase 3 完成后随时启动，不阻塞主线
                │
                ▼
         Phase 7 (集成测试)
```

**关键路径:** Phase 0 → 1 → 3 → 4 → 5 (后端 A 是瓶颈，β 和 α 的实现决定前端能否联调)

**可并行:**
- Phase 2(后端 B) 与 Phase 3(后端 A) 可由不同开发者并行，接口以 A-B 接口冻结文档为准
- Phase 6(小模型 α) 完全独立，在数据积累期随时启动
- 前端 P0-P2 可在后端 A 的 Web 层完成后立即开始联调
