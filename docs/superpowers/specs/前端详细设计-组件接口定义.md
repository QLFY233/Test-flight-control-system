## 2. 全局状态管理 (`state.js`)

单一全局 Store，发布订阅模式。所有组件通过 `store.subscribe(path, callback)` 订阅变化，通过 `store.set(path, value)` / `store.update(path, partial)` 修改。

### 2.1 状态结构

```typescript
// 以下用 TypeScript 类型表示法描述 JS 对象结构
// 实际实现是普通 JS Object，仅在注释中标注类型

interface GlobalState {
  // —— 连接状态 ——
  connection: {
    ws: 'connecting' | 'connected' | 'disconnected';
    wsErrorCount: number;         // 连续错误次数，用于指数退避
    sseBeta: 'idle' | 'connected' | 'disconnected';
    lastPingAt: number | null;    // 最近一次 ws pong 时间戳
  };

  // —— 无人机实时状态 (来自 WebSocket pose/telemetry) ——
  drone: {
    position: [number, number, number];    // [x, y, z] 米
    quaternion: [number, number, number, number]; // [w, x, y, z]
    velocity: [number, number, number];    // [vx, vy, vz] m/s
    accel: [number, number, number];       // [ax, ay, az] m/s²
    angularVelocity: [number, number, number]; // [wx, wy, wz] rad/s
    updatedAt: number;                     // 时间戳 ms
  };

  // —— 飞行状态 ——
  flight: {
    status: 'idle' | 'hovering' | 'planned' | 'executing' | 'completed' | 'aborted';
    mode: 'manual' | 'auto';              // 人类打字/语音操控 / 执行预设
    currentSegment: number;                // 当前执行到第几段 (0-based)
    totalSegments: number;                 // 总段数
    taskId: string | null;                 // 当前任务 ID
  };

  // —— 轨迹数据 (实时累积) ——
  trajectory: {
    flown: Array<{ t: number; pos: [number,number,number]; vel: [number,number,number] }>;
    planned: Array<{ pos: [number,number,number]; constraints?: object }>;
  };

  // —— β 界面数据 ——
  beta: {
    plan: FlightPlan | null;               // β 产出的飞行计划
    planStatus: 'none' | 'reviewing' | 'approved' | 'rejected';
  };

  // —— Chat 历史 (仅 β 对话,从后端 conversations 表拉取) ——
  chatHistory: {
    beta: ChatMessage[];
  };

  // —— 历史查询 ——
  history: {
    sessions: FlightSessionSummary[];
    loading: boolean;
  };

  // —— 场地和环境 ——
  field: {
    boundary: { x: [number,number]; y: [number,number]; z: [number,number] };
    obstacles: Obstacle[];
    home: { position: [number,number,number]; yaw: number };
  };
  environment: {
    current: EnvData | null;
    presets: Record<string, Partial<EnvData>>;
  };

  // —— UI 状态 ——
  ui: {
    activePage: 'overview' | 'beta' | 'alpha' | 'history' | 'settings';
    isMobile: boolean;
    keyboardInputFocused: boolean;          // Chat 输入框是否聚焦(用于阻止全局快捷键触发)

    // —— 右栏视图模式 ——
    viewMode: 1 | 2 | 3;                   // 1视图 / 2视图 / 3视图
    viewSlots: Array<{                     // 各槽位内容
      source: '3d' | 'video' | 'chart';
      label: string;
    }>;                                     // viewSlots.length === viewMode
    viewLocked: boolean;                    // 用户是否手动锁定了排列

    // —— 悬浮球 ——
    floatingBall: {
      visible: boolean;                    // 默认 true
      expanded: boolean;                   // 扇叶菜单是否展开
      shortcuts: ShortcutItem[];           // 当前快捷键列表
      editing: boolean;                    // 是否在编辑模式
    };
  };
}
```

### 2.2 Store API

```typescript
interface Store {
  // 订阅: path 用点号分隔，如 'drone.position'
  // 返回 unsubscribe 函数
  subscribe(path: string, callback: (newVal: any, oldVal: any) => void): () => void;

  // 订阅通配符: 'drone.*' 表示 drone 下任意字段变化
  subscribeWildcard(prefix: string, callback: (path: string, newVal: any, oldVal: any) => void): () => void;

  // 设置单个值
  set(path: string, value: any): void;

  // 批量更新 (触发一次通知)
  batch(updater: (state: GlobalState) => void): void;

  // 读取当前值
  get(path: string): any;

  // 读取整个 state (谨慎，只应在初始化或调试时用)
  getState(): GlobalState;
}
```

## 3. WebSocket 消息契约 (`ws.js`)

> 总体架构 §3.5: 所有跨进程 JSON 加 `schema_version` 字段。下方每条消息均带 `schema_version: number`,前端解析时按版本路由;未知版本忽略并告警。

### 3.1 ws 模块接口

```typescript
interface WsManager {
  // 初始化连接 (在 app.js 启动时调用)
  connect(url: string): void;

  // 断开连接 (页面卸载时)
  disconnect(): void;

  // 发送消息到后端 A
  send(type: string, payload: object): void;

  // 订阅特定类型的消息
  on(type: string, handler: (data: any) => void): () => void;

  // 连接状态
  getState(): 'connecting' | 'connected' | 'disconnected';
}
```

### 3.2 上行消息 (前端 → A)

```typescript
// 人类操控通过 Chat 文本(自然语言)发送,经 SSE/REST 路径,不走 WS keyboard 消息。
// 语音输入前端采集音频后发 voice_frame。

// 语音帧 (前端采 PCM 16k/16bit/单声道,每帧 ~1280 字节/40ms,base64 上行)
// 后端 A 转发至讯飞 STT(签名鉴权 + wpgs 动态修正),结果经 voice_stt_result 回填
interface VoiceFrameMessage {
  type: 'voice_frame';
  schema_version: number;
  data: string;      // base64 PCM 16k/16bit/单声道
  seq: number;       // 帧序列号
  ts: number;
}

// 断连重连后的状态同步请求
interface SyncMessage {
  type: 'sync';
  schema_version: number;
  ts: number;
}

// TTS 请求
interface TtsRequestMessage {
  type: 'tts_request';
  schema_version: number;
  text: string;
}
```

### 3.3 下行消息 (A → 前端)

```typescript
// 位姿更新 (10Hz)
interface PoseMessage {
  type: 'pose';
  schema_version: number;
  pos: [number, number, number];      // [x, y, z]
  quat: [number, number, number, number]; // [w, x, y, z]
  vel: [number, number, number];      // [vx, vy, vz]
  accel: [number, number, number];    // [ax, ay, az] m/s²
  angularVel: [number, number, number]; // [wx, wy, wz] rad/s
  ts: number;                          // 服务器时间戳
}

// 视频帧
interface VideoFrameMessage {
  type: 'video_frame';
  schema_version: number;
  data: string;                        // base64 JPEG 帧
  seq: number;
  ts: number;
}

// 任务进度
interface StatusMessage {
  type: 'status';
  schema_version: number;
  flightStatus: 'idle' | 'hovering' | 'planned' | 'executing' | 'completed' | 'aborted';
  mode: 'manual' | 'auto';
  currentSegment: number;
  totalSegments: number;
  taskId: string;
}

// TTS 音频下行
interface VoiceTtsMessage {
  type: 'voice_tts';
  schema_version: number;
  data: string;                        // base64 音频
  format: 'mp3';                       // 固定 mp3(讯飞 TTS lame 编码)
}

// STT 识别结果回填(流式,支持讯飞 wpgs 动态修正)
interface VoiceSttResultMessage {
  type: 'voice_stt_result';
  schema_version: number;
  text: string;        // 当前识别文本(中间帧实时回填输入框)
  is_final: boolean;   // 是否最终结果(松开/60s 上限时的末帧)
}

// α loop 产出的系统消息(无对话文字,仅推送执行中的 TrajectorySpec 摘要)
// 人类只与 β 对话,α 不输出文字;α_output 为系统消息(显示在 α 视图状态)
interface AlphaOutputMessage {
  type: 'alpha_output';
  schema_version: number;
  trajectory?: TrajectorySpec;          // 执行中的轨迹摘要(供 α 视图状态显示)
}

// 异常告警(B 监控程序检测,前端作为系统消息插入 β 对话流)
interface AlertMessage {
  type: 'alert';
  schema_version: number;
  level: 'info' | 'warn' | 'critical';
  code: string;
  detail: string;
  suggestion?: string;                  // β 给的处置建议(若有,人确认后执行)
}

// 求解器拒绝回告
interface RejectMessage {
  type: 'reject';
  schema_version: number;
  reason: string;
  segmentIndex: number;
  suggestedAction: string;
}

// 后端 A↔B 链路状态告警 (B 断连/恢复时由 A 转发, 总体架构 §3.6)
interface LinkStatusMessage {
  type: 'link_status';
  schema_version: number;
  link: 'A-B' | 'llm' | 'speech';     // 哪条链路
  state: 'ok' | 'degraded' | 'down';
  detail?: string;
}

// 位姿同步回复 (响应 sync 请求)
interface SyncResponseMessage {
  type: 'sync_response';
  schema_version: number;
  pose: PoseMessage['pos'];
  quat: PoseMessage['quat'];
  status: StatusMessage;
}

// TTS 下行格式固定为 base64 mp3(讯飞 TTS lame 编码),详见 §6.6a AudioPlayer

### 6.7 FlightPlanCard — 飞行计划卡片

> **关键区分（与总体架构 §2.1 三层翻译链对齐）:**
> - `FlightPlan` 是 **β 产出 → 人审核** 阶段的对象：自然语言 summary + 航点概要 + 约束建议。它是"人可读的计划"。
> - `TrajectorySpec` 是 **α 翻译 → 求解器** 阶段的对象：带 `task_id`/`schema_version` 的点序列 + 逐段约束（总体架构 §2.2 格式）。它是"机器执行用的点序列"。
> - β 产出 FlightPlan → 人审核通过 → **α 翻译**成 TrajectorySpec → 下发求解器。前端**不可**把 FlightPlan 直接当作 TrajectorySpec 使用。
> - 前端在 β 界面展示 FlightPlan；α 界面展示的是经 α 翻译后的 TrajectorySpec（由后端 SSE/WS 下发），两者数据结构不同。

```typescript

// FlightPlan: β 产出 → 人审核
interface FlightPlan {
  summary: string;                  // 自然语言概述（β 产出，人可读）
  segments: FlightPlanSegment[];    // 航点概要 + 约束建议
  globalConstraints?: {
    keepClearDistance?: number;
    ceiling?: number;
    floor?: number;
  };
}

// 渲染为结构化列表:
// ┌──────────────────────────────────────────────┐
// │ 📋 飞行计划                                  │
// │ 概述: 从起点起飞, 绕过柱子, 在 (3,2,1) 悬停...│
// │                                              │
// │ 段 1: (0,0,0.5) → (3,2,1.0)                 │
// │   约束: 速度 ≤ 1.2 m/s, 加速度 ≤ 2.0 m/s²   │
// │   避开: 柱子(方形)                            │
// │                                              │
// │ 段 2: (3,2,1.0) → (0,0,0.5) [返航]           │
// │   约束: 速度 ≤ 0.8 m/s                        │
// │                                              │
// │ [✓ 批准]  [✎ 修改]  [✗ 驳回]                │
// └──────────────────────────────────────────────┘
//
// "修改" 模式下每个坐标/约束字段变成可编辑输入框
// 点"批准"触发全局事件, 通知 BetaPage

interface FlightPlanSegment {
  waypoints: Array<{
    position: [number, number, number];
    yaw?: number;
    constraints?: Partial<SegmentConstraints>;
    returnHome?: boolean;
  }>;
  segmentConstraints?: Partial<SegmentConstraints>;
}

// TrajectorySpec: α 翻译 → 求解器（总体架构 §2.2 格式，与 FlightPlan 不同）
// 前端在 α 界面通过 WS/SSE 收到此结构用于展示"已注入执行中"的轨迹
interface TrajectorySpec {
  task_id: string;
  schema_version: number;
  global_constraints: {
    keep_clear_distance?: number;
    ceiling?: number;
    floor?: number;
    speed_max?: number;
    accel_max?: number;
    angular_velocity_max?: number;
    [key: string]: any;   // 可扩展约束
  };
  segments: Array<{
    waypoints: Array<{
      position: [number, number, number];
      yaw?: number;
      return_home?: boolean;
      constraints?: Record<string, number | object>;  // 点约束
    }>;
    segment_constraints?: Record<string, number | object>;  // 段约束
  }>;
}

### 6.12 ShortcutEditor — 快捷键编辑器

```typescript
interface ShortcutEditorProps {
  editingItem?: ShortcutItem;    // 正在编辑的快捷键, undefined = 新建
  onSave: (item: ShortcutItem) => void;
  onCancel: () => void;
  onRecord: () => void;           // 开始"记录操作"
  onStopRecord: () => void;       // 停止记录
  isRecording: boolean;           // 是否正在记录中
  recordedActions: RecordedAction[];  // 记录到的操作序列
}

// ShortcutItem 数据结构:
interface ShortcutItem {
  id: string;                     // 唯一 ID (crypto.randomUUID)
  name: string;                   // 快捷键显示名, 如 "上升 0.5m"
  icon: string;                   // 图标名, 如 'arrow-up'
  actions: ShortcutAction[];     // 操作序列 (一个或多个)
  color?: string;                 // 扇叶颜色 (可选)
}

type ShortcutAction =
  | { type: 'chat_message'; text: string }                      // 一键发送 Chat 消息(自然语言,α 翻译)
  | { type: 'delay'; ms: number }                               // 等待 N 毫秒
  | { type: 'sequence'; actions: ShortcutAction[] };            // 组合序列

// RecordedAction 数据结构:
interface RecordedAction {
  type: 'chat';
  text?: string;
  ts: number;                     // 相对时间戳 (ms, 从开始记录起)
  duration?: number;              // 操作持续时长
}

// 编辑器渲染 (小视图弹窗):
// ┌─────────────────────────────────────┐
// │  添加快捷键                  [✕]   │
// │                                     │
// │  名称: [________________]           │
// │  图标: [▼ 选择图标]                  │
// │                                     │
// │  操作:                               │
// │  ┌──────────────────────────────┐  │
// │  │ ① 发送 "上升 0.5 米" (Chat) [✕]│  │
// │  │ ② 等待 500ms         [✕]    │  │
// │  │ ③ 发送 "悬停" (Chat)    [✕]    │  │
// │  │ [ + 添加操作 ]               │  │
// │  └──────────────────────────────┘  │
// │                                     │
// │  [🎥 记录操作]←记录上次 Chat 发送    │
// │                                     │
// │  [  取消 ]   [  保存 ]              │
// └─────────────────────────────────────┘
//
// "🎥 记录操作" 功能:
//   1. 点击 [🎥 记录操作] → 编辑器关闭, 进入记录模式
//   2. 用户在 α 界面正常操作 (在 Chat 里打字发送指令)
//   3. 系统记录所有 Chat 发送到 recordedActions (含时间间隔)
//   4. 再次点击悬浮球 / 按 Esc → 停止记录
//   5. 回到 ShortcutEditor, recordedActions 填入操作列表
//   6. 用户可以增/删/改每个操作, 调整时间间隔
//
// 触发与取消:
//   - 快捷键触发即发送预存 Chat 消息(自然语言),由 α 翻译执行
//   - 多动作序列触发后, 第一个 action 发送前有 200ms 延迟
//   - 200ms 内按 Esc → 取消整个序列
//   - 界面提示: "已取消" (toast, 1s 消失)
//   - 单动作快捷键直接执行, 无取消窗口
//
// 小视图原则:
//   - ShortcutEditor 以"小弹窗"形式弹出 (480×640 max)

## 12. 异常与边界处理

| 场景 | 处理方式 | 涉及模块 |
|---|---|---|
| WebSocket 连接失败 | 指数退避 (1-2-4-8-16-30s cap)，更新 store.connection | ws.js |
| WebSocket 连接中收到消息 | 丢弃（状态不一致） | ws.js |
| WebSocket 断连后重连成功 | 发送 `sync` 消息请求全量状态 | ws.js |
| SSE 断流 | ChatPanel 显示"连接中断，正在重连"，自动重连 | sse.js |
| REST 请求失败 (4xx/5xx) | 静默重试 1 次，仍失败则显示 toast 错误提示 | api.js |
| Three.js WebGL 不支持 | 检测 `WebGLRenderingContext`，显示降级提示"3D 场景不可用" | Scene3D.js |
| 视频帧解码失败 | 跳过该帧，计数错误 | VideoPanel.js |
| 移动端旋转屏幕 | 监听 `orientationchange` / `resize`，重新布局 + ECharts resize | app.js |
| 浏览器标签页隐藏 | 暂停动画循环 (Page Visibility API)，切回时恢复 | Scene3D.js |
| 10Hz 数据积压 | 3D 插值追赶，不逐帧回放；ECharts 丢弃旧点 | 各模块 |
| 悬浮球快捷键与 Chat 输入冲突 | 仅在 `store.ui.keyboardInputFocused === false` 时触发快捷键 | AlphaPage.js |
| 页面关闭/刷新 | beforeunload 时断开 WebSocket 和 SSE | app.js |
| 求解器 reject 某段 (ws `reject`) | toast 告警 + ChatPanel 追加"段 N 被拒绝: {reason}",α 重新翻译或告警人 | ChatPanel.js / app.js |
| A↔B 链路断连 (ws `link_status` state=down) | StatusBar 显示 B 链路红色告警 + ConnectionOverlay 提示"飞控失联,无人机已安全悬停" | StatusBar.js / ConnectionOverlay.js |
| 大模型/语音 API 失败 (ws `link_status` state=down link=llm/speech) | StatusBar 对应项告警 + ChatPanel 提示"大模型不可用,维持悬停" | StatusBar.js / ChatPanel.js |
| α 安全悬停 (status `flightStatus`='hovering') | BottomBar 显示"安全悬停中"(黄色),飞控总有合法期望值 | BottomBar.js |

  ├── 2. 初始化右栏视图
  │     ├── viewModeSelector / scene3D / droneModel / charts ...
  │
  ├── 3. 悬浮球快捷键

### 13.2 人类操控流程(经 β→α 调度)

> 人类只与 β 对话。操控路径:人在 β Chat 打字 → β 调 forward/propose → α 翻译→求解器。

```
用户在 β 界面 Chat 输入框打字 "上升 0.5 米"
  │
  ├── 1. POST /api/chat/beta { message: "上升 0.5 米" }
  ├── 2. β 理解意图 → 调 forward_last_human_message (或 propose_to_alpha → 人确认)
  ├── 3. 指令经总线入 α 队列 → α loop 下周期翻译成 TrajectorySpec
  ├── 4. 经总线发 solver → 求解器生成连续轨迹 → 假无人机
  │
  └── 5. 位姿更新 → WS pose → 前端刷新 3D/图表
        alpha_output 系统消息 → α 视图显示当前执行进度
```

### 13.3 β 生成 → 批准 → α 翻译 → α 执行

> 与总体架构 §2.1 三层翻译链对齐：β 产出 FlightPlan（人审核）→ α 翻译成 TrajectorySpec（求解器执行）。前端**不**把 FlightPlan 直接当 TrajectorySpec。

```
用户在 β 界面描述任务
  │
  ├── 1. 用户输入 "从起点起飞，绕柱子一周，回到起点"
  ├── 2. POST /api/chat/beta { message } → SSE 流式回复
  ├── 3. ChatPanel 逐字渲染 β 回复
  ├── 4. β 调工具 get_field_map() → fieldMap 更新标注
  ├── 5. β 产出 FlightPlan（自然语言 summary + 航点概要）→ FlightPlanCard 渲染
  │
  ├── 6. 用户点 [批准]
  │     ├── store.beta.planStatus = 'approved'
  │     ├── POST /api/plan/approve { plan: FlightPlan }   ← 发的是 FlightPlan
  │     └── router.navigate('/alpha')  — 跳转到操控界面
  │
  └── 7. α 界面中:
        ├── 后端 A 的 α agent 将 FlightPlan 翻译成 TrajectorySpec
        ├── α 经 WS alpha_output 系统消息下发 TrajectorySpec → store.trajectory.planned 更新
        ├── trajectoryLine.setPlanned(TrajectorySpec.segments 的航点)
        └── 自动执行(或等待 β 发 forward/人确认执行)
```

## 16. 全局事件总线

除了 Store 的状态管理，组件间还需要事件总线来处理一次性动作：

```typescript
interface EventBus {
  // 路由 & 通知
  on(event: 'navigate', handler: (page: string) => void): void;
  on(event: 'show-toast', handler: (msg: { text: string; level: 'info'|'warn'|'error' }) => void): void;

  // 飞行计划
  on(event: 'plan-approved', handler: (plan: FlightPlan) => void): void;

  // 安全
  on(event: 'emergency-hover', handler: () => void): void;

  // 悬浮球快捷键
  on(event: 'shortcut-triggered', handler: (shortcut: ShortcutItem) => void): void;
  on(event: 'shortcut-record-start', handler: () => void): void;
  on(event: 'shortcut-record-stop', handler: (actions: RecordedAction[]) => void): void;

  // 视图模式
  on(event: 'view-mode-changed', handler: (mode: 1 | 2 | 3) => void): void;
  on(event: 'view-source-changed', handler: (slotIndex: number, source: string) => void): void;

  emit(event: string, data?: any): void;
  off(event: string, handler: Function): void;
}

// 使用场景:
//   - 'navigate'              → router.navigate()
//   - 'show-toast'            → 显示短暂提示 (REST 失败等)
//   - 'plan-approved'         → BetaPage 通知全局, 触发跳转到 α
//   - 'emergency-hover'       → 任意页面触发悬停 (全局快捷键)
//   - 'shortcut-triggered'    → FloatingBall → AlphaPage 执行操作序列
//   - 'shortcut-record-start' → 开始记录用户操作 (ShortcutEditor)
//   - 'shortcut-record-stop'  → 停止记录, 返回操作序列
//   - 'view-mode-changed'     → ViewModeSelector → 更新 store.ui.viewMode
//   - 'view-source-changed'   → ViewPanel 源切换 → 重新挂载渲染器
```
